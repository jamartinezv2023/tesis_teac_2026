% Estilo uniforme para tablas (APA-friendly)
\newcommand{\APAtablesetup}{%
  \small
  \setlength{\tabcolsep}{4pt}%
  \renewcommand{\arraystretch}{1.25}%
}

% Texto automático para tablas largas
\newcommand{\APAtablecontinued}{\footnotesize\textit{Continúa en la siguiente página.}}

% =========================
% Entorno reusable: longtable estilo APA
% Uso:
% \begin{APAlongtable}{<cols>}{<caption>}{<label>}
%   Encabezado... \\ \midrule
%   \endfirsthead ... \endhead ... \endfoot ... \endlastfoot
%   Filas...
% \end{APAlongtable}
% =========================
\newenvironment{APAlongtable}[3]{%
  \APAtablesetup
  \begin{center}
  \begin{longtable}{#1}
  \caption{#2}\label{#3}\\
  \toprule
}{%
  \bottomrule
  \end{longtable}
  \end{center}
}
